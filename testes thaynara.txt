
import psycopg2
from dotenv import load_dotenv
import os

load_dotenv()

def conectar():

    return psycopg2.connect(
        host=os.getenv("DB_HOST"),
        database=os.getenv("DB_NAME"),
        user=os.getenv("DB_USER"),
        password=os.getenv("DB_PASSWORD"),
        port=os.getenv("DB_PORT")
    )

from conexao import conectar


def cadastrar_usuario():

    nome = input("Nome: ")
    email = input("Email: ")
    senha = input("Senha: ")
    matricula = input("Matricula: ")
    telefone = input("Telefone: ")

    print("\nTipos:")
    print("1 - Aluno")
    print("2 - Professor")
    print("3 - Coordenador")
    print("4 - Pesquisador")
    print("5 - Monitor")
    print("6 - Administrador")

    tipo = int(input("Tipo: "))

    conn = conectar()
    cur = conn.cursor()

    cur.execute("""
        INSERT INTO usuarios
        (nome,email,senha,matricula,telefone,id_tipo_usuario)
        VALUES (%s,%s,%s,%s,%s,%s)
    """,
    (nome,email,senha,matricula,telefone,tipo))

    conn.commit()

    print("\nUsuário cadastrado!")

    cur.close()
    conn.close()


def listar_usuarios():

    conn = conectar()
    cur = conn.cursor()

    cur.execute("""
        SELECT
            id_usuario,
            nome,
            email
        FROM usuarios
        ORDER BY id_usuario
    """)

    usuarios = cur.fetchall()

    print("\n===== USUÁRIOS =====")

    for u in usuarios:
        print(u)

    cur.close()
    conn.close()
